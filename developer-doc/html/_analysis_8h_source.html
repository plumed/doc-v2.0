130 unsigned Analysis::getNumberOfDataPoints() const {
131  if( !reusing_data ){
132  plumed_dbg_assert( data.size()==logweights.size() );
133  return data.size();
134  } else {
135  return mydatastash->getNumberOfDataPoints();
136  }

139 inline
140 void Analysis::getDataPoint( const unsigned& idata, std::vector<double>& point, double& weight ) const {
141  if( !reusing_data ){
142  plumed_dbg_assert( idata<weights.size() && point.size()==getNumberOfArguments() );
143  for(unsigned i=0;i<point.size();++i) point[i]=data[idata][i];
144  weight=weights[idata];
145  } else {
146  return mydatastash->getDataPoint( idata, point, weight );
147  }
148 }
