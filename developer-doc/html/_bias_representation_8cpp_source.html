118 void BiasRepresentation::pushKernel( IFile *ifile ){
119  KernelFunctions *kk;
120  // here below the reading of the kernel is completely hidden
121  if(histosigma.size()==0){
122  ifile->allowIgnoredFields();
123  kk=KernelFunctions::read(ifile,names) ;
124  }else{
125  // when doing histogram assume gaussian with a given diagonal sigma
126  // and neglect all the rest
127  kk=readFromPoint(ifile) ;
128  }
129  hills.push_back(kk);
130  // the bias factor is not something about the kernels but
131  // must be stored to keep the bias/free energy duality
132  string dummy; double dummyd;
133  if(ifile->FieldExist("biasf")){
134  ifile->scanField("biasf",dummy);
135  Tools::convert(dummy,dummyd);
136  }else{dummyd=1.0;}
137  biasf.push_back(dummyd);
138  // the domain does not pertain to the kernel but to the values here defined
139  string mins,maxs,minv,maxv,mini,maxi;mins="min_";maxs="max_";
140  for(unsigned i=0 ; i<ndim; i++){
141  if(values[i]->isPeriodic()){
142  ifile->scanField(mins+names[i],minv);
143  ifile->scanField(maxs+names[i],maxv);
144  // verify that the domain is correct
145  values[i]->getDomain(mini,maxi);
146  plumed_massert(mini==minv,"the input periodicity in hills and in value definition does not match" );
147  plumed_massert(maxi==maxv,"the input periodicity in hills and in value definition does not match" );
148  }
149  }
150  // if grid is defined then it should be added on the grid
151  //cerr<<"now with "<<hills.size()<<endl;
152  if(hasgrid){
153  vector<unsigned> nneighb=kk->getSupport(BiasGrid_->getDx());
154  vector<unsigned> neighbors=BiasGrid_->getNeighbors(kk->getCenter(),nneighb);
155  vector<double> der(ndim);
156  vector<double> xx(ndim);
157  if(mycomm.Get_size()==1){
158  for(int i=0;i<neighbors.size();++i){
159  unsigned ineigh=neighbors[i];
160  for(int j=0;j<ndim;++j){der[j]=0.0;}
161  BiasGrid_->getPoint(ineigh,xx);
162  // assign xx to a new vector of values
163  for(int j=0;j<ndim;++j){values[j]->set(xx[j]);}
164  double bias=kk->evaluate(values,der,true);
165  if(rescaledToBias){
166  double f=(biasf.back()-1.)/(biasf.back());
167  bias*=f;
168  for(int j=0;j<ndim;++j){der[j]*=f;}
169  }
170  BiasGrid_->addValueAndDerivatives(ineigh,bias,der);
171  }
172  } else {
173  unsigned stride=mycomm.Get_size();
174  unsigned rank=mycomm.Get_rank();
175  vector<double> allder(ndim*neighbors.size(),0.0);
176  vector<double> allbias(neighbors.size(),0.0);
177  vector<double> tmpder(ndim);
178  for(unsigned i=rank;i<neighbors.size();i+=stride){
179  unsigned ineigh=neighbors[i];
180  BiasGrid_->getPoint(ineigh,xx);
181  for(int j=0;j<ndim;++j){values[j]->set(xx[j]);}
182  allbias[i]=kk->evaluate(values,tmpder,true);
183  if(rescaledToBias){
184  double f=(biasf.back()-1.)/(biasf.back());
185  allbias[i]*=f;
186  for(int j=0;j<ndim;++j){tmpder[j]*=f;}
187  }
188  // this solution with the temporary vector is rather bad, probably better to take
189  // a pointer of double as it was in old gaussian
190  for(int j=0;j<ndim;++j){ allder[ndim*i+j]=tmpder[j];tmpder[j]=0.;}
191  }
192  mycomm.Sum(&allbias[0],allbias.size());
193  mycomm.Sum(&allder[0],allder.size());
194  for(unsigned i=0;i<neighbors.size();++i){
195  unsigned ineigh=neighbors[i];
196  for(unsigned j=0;j<ndim;++j){der[j]=allder[ndim*i+j];}
197  BiasGrid_->addValueAndDerivatives(ineigh,allbias[i],der);
198  }
199  }
200  }

209 void BiasRepresentation::getMinMaxBin(vector<double> &vmin, vector<double> &vmax, vector<unsigned> &vbin){
210  vector<double> ss,cc,binsize;
211  vmin.clear();vmin.resize(ndim,10.e20);
212  vmax.clear();vmax.resize(ndim,-10.e20);
213  vbin.clear();vbin.resize(ndim);
214  binsize.clear();binsize.resize(ndim,10.e20);
215  int ndiv=10; // adjustable parameter: division per support
216  for(unsigned i=0;i<hills.size();i++){
217  if(histosigma.size()!=0){
218  ss=histosigma;
219  }else{
220  ss=hills[i]->getContinuousSupport();
221  }
222  cc=hills[i]->getCenter();
223  for(unsigned j=0;j<ndim;j++){
224  double dmin=cc[j]-ss[j];
225  double dmax=cc[j]+ss[j];
226  double ddiv=ss[j]/double(ndiv);
227  if(dmin<vmin[j])vmin[j]=dmin;
228  if(dmax>vmax[j])vmax[j]=dmax;
229  if(ddiv<binsize[j])binsize[j]=ddiv;
230  }
231  }
232  for(unsigned j=0;j<ndim;j++){
233  // reset to periodicity
234  if(values[j]->isPeriodic()){
235  double minv,maxv;
236  values[j]->getDomain(minv,maxv);
237  if(minv>vmin[j])vmin[j]=minv;
238  if(maxv<vmax[j])vmax[j]=maxv;
239  }
240  vbin[j]=static_cast<unsigned>(ceil((vmax[j]-vmin[j])/binsize[j]) );
241  }
242 }
