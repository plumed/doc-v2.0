33 void ActionWithVessel::registerKeywords(Keywords& keys){
34  keys.add("optional","TOL","this keyword can be used to speed up your calculation. When accumulating sums in which the individual "
35  "terms are numbers inbetween zero and one it is assumed that terms less than a certain tolerance "
36  "make only a small contribution to the sum. They can thus be safely ignored as can the the derivatives "
37  "wrt these small quantities.");
38  keys.reserve("hidden","NL_TOL","this keyword can be used to speed up your calculation. It must be used in conjuction with the TOL "
39  "keyword and the value for NL_TOL must be set less than the value for TOL. This keyword ensures that "
40  "quantities, which are much less than TOL and which will thus not added to the sums being accumulated "
41  "are not calculated at every step. They are only calculated when the neighbor list is updated.");
42  keys.addFlag("SERIAL",false,"do the calculation in serial. Do not parallelize");
43  keys.add( vesselRegister().getKeywords() );
44 }

93 void ActionWithVessel::readVesselKeywords(){
94  // Loop over all keywords find the vessels and create appropriate functions
95  for(unsigned i=0;i<keywords.size();++i){
96  std::string thiskey,input; thiskey=keywords.getKeyword(i);
97  // Check if this is a key for a vessel
98  if( vesselRegister().check(thiskey) ){
99  // If the keyword is a flag read it in as a flag
100  if( keywords.style(thiskey,"flag") ){
101  bool dothis; parseFlag(thiskey,dothis);
102  if(dothis) addVessel( thiskey, input );
103  // If it is numbered read it in as a numbered thing
104  } else if( keywords.numbered(thiskey) ) {
105  parse(thiskey,input);
106  if(input.size()!=0){
107  addVessel( thiskey, input );
108  } else {
109  for(unsigned i=1;;++i){
110  if( !parseNumbered(thiskey,i,input) ) break;
111  std::string ss; Tools::convert(i,ss);
112  addVessel( thiskey, input, i );
113  input.clear();
114  }
115  }
116  // Otherwise read in the keyword the normal way
117  } else {
118  parse(thiskey, input);
119  if(input.size()!=0) addVessel(thiskey,input);
120  }
121  input.clear();
122  }
123  }
124 
125  // Make sure all vessels have had been resized at start
126  if( functions.size()>0 ) resizeFunctions();
127 }

171 void ActionWithVessel::runAllTasks(){
172  plumed_massert( read, "you must have a call to readVesselKeywords somewhere" );
173  unsigned stride=comm.Get_size();
174  unsigned rank=comm.Get_rank();
175  if(serial){ stride=1; rank=0; }
176 
177  // Make sure jobs are done
178  doJobsRequiredBeforeTaskList();
179 
180  for(unsigned i=rank;i<taskList.getNumberActive();i+=stride){
181  // Store the task we are currently working on
182  current=taskList[i];
183  // Calculate the stuff in the loop for this action
184  performTask(i);
185  // Weight should be between zero and one
186  plumed_dbg_assert( thisval[1]>=0 && thisval[1]<=1.0 );
187 
188  // Check for conditions that allow us to just to skip the calculation
189  // the condition is that the weight of the contribution is low
190  // N.B. Here weights are assumed to be between zero and one
191  if( thisval[1]<tolerance ){
192  // Clear the derivatives
193  clearAfterTask();
194  // Deactivate task if it is less than the neighbor list tolerance
195  if( thisval[1]<nl_tolerance && contributorsAreUnlocked ) deactivate_task();
196  continue;
197  }
198 
199  // Now calculate all the functions
200  // If the contribution of this quantity is very small at neighbour list time ignore it
201  // untill next neighbour list time
202  if( !calculateAllVessels() && contributorsAreUnlocked ) deactivate_task();
203  }
204  finishComputations();
205 }
