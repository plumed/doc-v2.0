39 Calculate Debye-Huckel interaction energy among GROUPA and GROUPB.
40 
41 This collective variable can be used to analyze or induce electrostatically driven reactions \cite do13jctc.

76  /*
77  Global constants in SI unit used in this calculation:
78  N_A = 6.0221412927 * 10^(23) mol^(-1) : Avogadro number
79  q = 1.60217656535 * 10^(-19) C : proton charge
80  e_0 = 8.854187817620 * 10^(-12) C^2/(N*m^2) : vacuum's dielectric constant
81  k_B = 1.380648813 * 10^(-23) N*m/K : Boltzmann constant
82  In SI unit, Debye Huckel CV is defined as:
83  DHen = \sum_i\sum_j (q_i*q_j*q^2*N_A)/(4*pi*eps*e_0) * exp(-k*|f_ij|)/(|f_ij|)
84  + \sum_i\sum_j (q_i*q_j*q^2*N_A)/(4*pi*epp*e_0) * (1/|r_ij| - 1/|f_ij|)
85  = (q^2*N_A)/(4*pi*e_0) * \sum_i\sum_j q_i*q_j * (exp(-k*|f_ij|)/(eps*|f_ij|) + 1/epp*(1/|r_ij| - 1/|f_ij|))
86  (in which |f_ij| = \sqrt(|r_ij|^2+\sigma_i*\sigma_j*exp(-|r_ij|^2/4*\sigma_i*\sigma_j)),
87  \sigma_i and \sigma_j are the effective Born radius.)
88  For an efficient calculation, we group constants and variables into groups:
89  constant = (q^2*N_A)/(4*pi*e_0)
90  tmp = 1/eps*exp(-k*|f_ij|)/(|f_ij|) + 1/epp*(1/|r_ij| - 1/|f_ij|)
91 
92  To speed up the loop calculation, constant can be modified as followed:
93  constant= (q^2*N_A)/(4*pi*e_0*10^(-9))*10^(-3) (kJ/mol)
94  = ((1.60217656535*10^(-19))^2*6.0221412927*10^(23)*10^(-3))/(4*3.14159265*8.854187817620*10^(-12)*10^(-9))
95  = 138.935458111 (kJ/mol)
96 
97  */

99 DHEnergy::DHEnergy(const ActionOptions&ao):
100 Action(ao),
101 CoordinationBase(ao),
102 k(0.0),
103 constant(0.0)
104 {
105  double I,T;
106  parse("I",I);
107  parse("TEMP",T);
108  parse("EPSILON",epsilon);
109  checkRead();
110  if( plumed.getAtoms().usingNaturalUnits() ) error("DHENERGY cannot be used for calculations performed with natural units");
111  constant=138.935458111/atoms.getUnits().getEnergy()/atoms.getUnits().getLength();
112  k=sqrt(I/(epsilon*T))*502.903741125*atoms.getUnits().getLength();
113  checkRead();
114  log<<" with solvent dielectric constant "<<epsilon<<"\n";
115  log<<" at temperature "<<T<<" K\n";
116  log<<" at ionic strength "<<I<< "M\n";
117  log<<" these parameters correspond to a screening length of "<<(1.0/k)<<"\n";
118  log<<" Bibliography "<<plumed.cite("Do, Carloni, Varani and Bussi, J. Chem. Theory Comput. 9, 1720 (2013)")<<" \n";
119 }

121 double DHEnergy::pairing(double distance,double&dfunc,unsigned i,unsigned j)const{
122  if(getAbsoluteIndex(i)==getAbsoluteIndex(j)){
123  dfunc=0.0;
124  return 0.0;
125  }
126  double invdistance=1.0/distance;
127  double tmp=exp(-k*distance)*invdistance*constant*getCharge(i)*getCharge(j)/epsilon;
128  double dtmp=-(k+invdistance)*tmp;
129  dfunc=dtmp*invdistance;
130  return tmp;
131 }
