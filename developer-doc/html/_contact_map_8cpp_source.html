36 /*
37 Calculate the distances between a number of pairs of atoms and transform each distance by a switching function.
38 
39 \par Examples
40 
41 The following example calculates switching functions based on the distances between atoms
42 1 and 2, 3 and 4 and 4 and 5. The values of these three switching functions are then output
43 to a file named colvar.
44 
45 \verbatim
46 CONTACTMAP ATOMS1=1,2 ATOMS2=3,4 ATOMS3=4,5 ATOMS4=5,6 SWITCH=(RATIONAL R_0=1.5) LABEL=f1
47 PRINT ARG=f1.* FILE=colvar
48 \endverbatim
49 (See also \ref PRINT)
50 
51 */

84 ContactMap::ContactMap(const ActionOptions&ao):
85 PLUMED_COLVAR_INIT(ao),
86 pbc(true),
87 dosum(false)
88 {
89  parseFlag("SUM",dosum);
90  bool nopbc=!pbc;
91  parseFlag("NOPBC",nopbc);
92  pbc=!nopbc;
93 
94  // Read in the atoms
95  std::vector<AtomNumber> t, ga_lista, gb_lista;
96  for(int i=1;;++i ){
97  parseAtomList("ATOMS", i, t );
98  if( t.empty() ) break;
99 
100  if( t.size()!=2 ){
101  std::string ss; Tools::convert(i,ss);
102  error("ATOMS" + ss + " keyword has the wrong number of atoms");
103  }
104  ga_lista.push_back(t[0]); gb_lista.push_back(t[1]);
105  t.resize(0);
106 
107  // Add a value for this contact
108  std::string num; Tools::convert(i,num);
109  if(!dosum) addComponentWithDerivatives("contact"+num); componentIsNotPeriodic("contact"+num);
110  }
111  // Create neighbour lists
112  nl= new NeighborList(ga_lista,gb_lista,true,pbc,getPbc());
113  requestAtoms(nl->getFullAtomList());
114 
115  // Read in switching functions
116  std::string errors; sfs.resize( ga_lista.size() ); unsigned nswitch=0;
117  for(unsigned i=0;i<ga_lista.size();++i){
118  std::string num, sw1; Tools::convert(i+1, num);
119  if( !parseNumbered( "SWITCH", i+1, sw1 ) ) break;
120  nswitch++; sfs[i].set(sw1,errors);
121  if( errors.length()!=0 ) error("problem reading SWITCH" + num + " keyword : " + errors );
122  }
123  if( nswitch==0 ){
124  std::string sw; parse("SWITCH",sw);
125  if(sw.length()==0) error("no switching function specified use SWITCH keyword");
126  for(unsigned i=0;i<ga_lista.size();++i){
127  sfs[i].set(sw,errors);
128  if( errors.length()!=0 ) error("problem reading SWITCH keyword : " + errors );
129  }
130  } else if( nswitch!=sfs.size() ){
131  std::string num; Tools::convert(nswitch+1, num);
132  error("missing SWITCH" + num + " keyword");
133  }
134

136  for(unsigned i=0;i<sfs.size();++i){
137  log.printf(" The %dth contact is calculated from atoms : %d %d. Inflection point of switching function is at %s\n", i+1, ga_lista[i].serial(), gb_lista[i].serial() , ( sfs[i].description() ).c_str() );
138  }

152 void ContactMap::calculate(){
153 
154  double ncoord=0., coord;
155  Tensor virial;
156  std::vector<Vector> deriv(getNumberOfAtoms());
157 
158  for(unsigned int i=0;i<nl->size();i++) { // sum over close pairs
159  Vector distance;
160  unsigned i0=nl->getClosePair(i).first;
161  unsigned i1=nl->getClosePair(i).second;
162  if(pbc){
163  distance=pbcDistance(getPosition(i0),getPosition(i1));
164  } else {
165  distance=delta(getPosition(i0),getPosition(i1));
166  }
167 
168  double dfunc=0.;
169  coord = sfs[i].calculate(distance.modulo(), dfunc);
170  if( !dosum ) {
171  Value* val=getPntrToComponent( i );
172  setAtomsDerivatives( val, i0, (-dfunc)*distance );
173  setAtomsDerivatives( val, i1, dfunc*distance );
174  setBoxDerivatives( val, (-dfunc)*Tensor(distance,distance) );
175  val->set(coord);
176  } else {
177  deriv[i0] = deriv[i0] + (-dfunc)*distance ;
178  deriv[i1] = deriv[i1] + dfunc*distance ;
179  virial=virial+(-dfunc)*Tensor(distance,distance);
180  ncoord += coord;
181  }
182  }
183 
184  if( dosum ){
185  for(unsigned i=0;i<deriv.size();++i) setAtomsDerivatives(i,deriv[i]);
186  setValue (ncoord);
187  setBoxDerivatives (virial);
188  }
189 }
190 
191 }
