33 //+PLUMEDOC INTERNAL reweighting
34 /*
35 Calculate free energies from a biassed/higher temperature trajectory.
36 
37 We can use our knowledge of the Boltzmann distribution in the cannonical ensemble to reweight the data
38 contained in trajectories. Using this procedure we can take trajectory at temperature \f$T_1\f$ and use it to
39 extract probabilities at a different temperature, \f$T_2\f$, using:
40 
41 \f[
42 P(s',t) = \frac{ \sum_{t'}^t \delta( s(x) - s' ) \exp\left( +( \left[\frac{1}{T_1} - \frac{1}{T_2}\right] \frac{U(x,t')}{k_B} \right) }{ \sum_t'^t \exp\left( +\left[\frac{1}{T_1} - \frac{1}{T_2}\right] \frac{U(x,t')}{k_B} \right) }
43 \f]
44 
45 where \f$U(x,t')\f$ is the potential energy of the system. Alternatively, if a static or pseudo-static bias \f$V(x,t')\f$ is acting on
46 the system we can remove this bias and get the unbiased probability distribution using:
47 
48 \f[
49 P(s',t) = \frac{ \sum_{t'}^t \delta( s(x) - s' ) \exp\left( +\frac{V(x,t')}{k_B T} \right) }{ \sum_t'^t \exp\left( +\frac{V(x,t')}{k_B T} \right) }
50 \f]
51 
52 */
53 //+ENDPLUMEDOC

146  if(nomemory) log.printf(" doing a separate analysis for each block of data\n");
147  parseFlag("WRITE_CHECKPOINT",write_chq);

149  write_chq=false;

154  std::string filename = getName() + "_" + getLabel() + ".chkpnt";

160  readDataFromFile( filename );
161  // Setup the restart file (append mode)
162  if( write_chq ) rfile.open( filename.c_str() ); // In append mode automatically because of restart
163  // Run the analysis if we stoped in the middle of it last time
164  log.printf(" restarting analysis with %u points read from restart file\n",idata);
165  } else if( write_chq ){
166  // Setup the restart file (delete any old one)
167  rfile.open( filename.c_str() ); // In overwrite mode automatically because there is no restart
168  }
169  if( write_chq ){
170  rfile.addConstantField("old_normalization");
171  for(unsigned i=0;i<getNumberOfArguments();++i) rfile.setupPrintValue( getPntrToArgument(i) );
172  }
173  }

176 void Analysis::readDataFromFile( const std::string& filename ){
177  double tstep, oldtstep; IFile ifile;
178  if( !ifile.FileExist(filename) ) error("failed to find required restart file " + filename );
179  ifile.open(filename);
180 
181  bool first=true;
182  while(ifile.scanField("time",tstep)){
183  if( !first && ((tstep-oldtstep) - getStride()*plumed.getAtoms().getTimeStep())>plumed.getAtoms().getTimeStep() ){
184  error("frequency of data storage in " + filename + " is not equal to frequency of data storage plumed.dat file");
185  }
186  for(unsigned j=0;j<getNumberOfArguments();++j){
187  ifile.scanField( getPntrToArgument(j)->getName(), data[idata][j] );
188  }
189  ifile.scanField("log_weight",logweights[idata]);
190  ifile.scanField("old_normalization",old_norm);
191  ifile.scanField();
192  idata++; first=false; oldtstep=tstep;
193  }
194  ifile.close();
195  if(old_norm>0) firstAnalysisDone=true;
196 }

211 void Analysis::calculate(){
212  // Don't store the first step (also don't store if we are getting data from elsewhere)
213  if( getStep()==0 || reusing_data ) return;
214  // This is used when we have a full quota of data from the first run
215  if( idata==logweights.size() ) return;
216 
217  // Retrieve the bias
218  double bias=0.0; for(unsigned i=0;i<biases.size();++i) bias+=biases[i]->get();
219 
220  double ww=0;
221  if(needeng){
222  double energy=plumed.getAtoms().getEnergy()+bias;
223  // Reweighting because of temperature difference
224  ww=-( (1.0/rtemp) - (1.0/simtemp) )*(energy+bias) / plumed.getAtoms().getKBoltzmann();
225  }
226  // Reweighting because of biases
227  if( !biases.empty() ) ww += bias/( plumed.getAtoms().getKBoltzmann()*simtemp );
228 
229  if(single_run){
230  // Get the arguments and store them in a vector of vectors
231  for(unsigned i=0;i<getNumberOfArguments();++i) args[i]=getArgument(i);
232  data.push_back( args );
233  logweights.push_back(ww);
234  } else {
235  // Get the arguments and store them in a vector of vectors
236  for(unsigned i=0;i<getNumberOfArguments();++i) data[idata][i]=getArgument(i);
237  logweights[idata] = ww;
238  }
239  // Write data to checkpoint file
240  if( write_chq ) {
241  rfile.printField("time",getTime()); rfile.printField("old_normalization",old_norm);
242  for(unsigned i=0;i<getNumberOfArguments();++i) rfile.printField( getPntrToArgument(i), getArgument(i) );
243  rfile.printField("log_weight",logweights[idata]); rfile.printField();
244  }
245  // Increment data counter
246  idata++;
247 }

253 void Analysis::finalizeWeights( const bool& ignore_weights ){
254  // Check that we have the correct ammount of data
255  if( !reusing_data && idata!=logweights.size() ) error("something has gone wrong. Am trying to run analysis but I don't have sufficient data");
256  if( weights.size()!=logweights.size() ) weights.resize( logweights.size() );
257 
258  norm=0; // Reset normalization constant
259  if( ignore_weights ){
260  for(unsigned i=0;i<logweights.size();++i){
261  weights[i]=1.0; norm+=1.0;
262  }
263  } else if( nomemory ){
264  // Find the maximum weight
265  double maxweight=logweights[0];
266  for(unsigned i=1;i<getNumberOfDataPoints();++i){
267  if(logweights[i]>maxweight) maxweight=logweights[i];
268  }
269  // Calculate normalization constant
270  for(unsigned i=0;i<logweights.size();++i){
271  norm+=exp( logweights[i]-maxweight );
272  }
273  // Calculate weights (no memory)
274  for(unsigned i=0;i<logweights.size();++i){
275  weights[i]=exp( logweights[i]-maxweight );
276  }
277  // Calculate normalized weights (with memory)
278  } else {
279  // Calculate normalization constant
280  for(unsigned i=0;i<logweights.size();++i){
281  norm+=exp( logweights[i] );
282  }
283  if( !firstAnalysisDone ) old_norm=1.0;
284  // Calculate weights (with memory)
285  for(unsigned i=0;i<logweights.size();++i){
286  weights[i] = exp( logweights[i] ) / old_norm;
287  }
288  if( !firstAnalysisDone ) old_norm=0.0;
289  }
290 }

292 void Analysis::runAnalysis(){
293 
294  // close the restart file so it is flushed
295  if( write_chq ) rfile.close();
296 
297  // Note : could add multiple walkers here - simply read in the data from all
298  // other walkers here if we are writing the check points.
299 
300  // Calculate the final weights from the log weights
301  if( !reusing_data ){
302  finalizeWeights( ignore_reweight );
303  } else {
304  mydatastash->finalizeWeights( ignore_reweight );
305  norm=mydatastash->retrieveNorm();
306  }
307  // And run the analysis
308  performAnalysis(); idata=0;
309  // Update total normalization constant
310  old_norm+=norm; firstAnalysisDone=true;
311 
312  // Delete the checkpoint file
313  if( write_chq ){
314  std::string filename = getName() + "_" + getLabel() + ".chkpnt";
315  // If we are running more than one calculation only reopen the restart file
316  if( !single_run ) rfile.open( filename.c_str(), "w+" );
317  }
318 }
